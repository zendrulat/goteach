import (

	"net/http"

IMPORTS FOR "PACKAGE" CORE/THIRD PARTY 

var temps *template.Template
VARIABLE "temps" TO STORE CORE TEMPLATE WITH A * POINTER TO IT 

	//init for running first
func init() {

	temps = template.Must(template.ParseGlob("templates/*.html"))
}
VARIABLE "temps" IS ALSO GETTING THE TEMPLATE METHOD TO PARSE FILES IN A FOLDER 

	//required func main()
func main() {

	//basically different url segments and the methods you made
	http.HandleFunc("/", index)
	http.HandleFunc("/golang", golang)
	http.HandleFunc("/linux", linux)
	http.HandleFunc("/resources", resources)

	//the server port you are using
	http.ListenAndServe(":8080", nil)

}
THE IMPORTED net/http "http" WITH ITS METHOD "HandleFunc()"
THE "/" IS THE URL PATH, "index" IS THE EXCUTING METHOD WHEN YOU GO TO THAT PATH 

func index(res http.ResponseWriter, req *http.Request) {
	//actual file
	err := temps.ExecuteTemplate(res, "index.html", nil)
	//checking err
	if err != nil {
		log.Fatalln("template didn't execute: ", err)
	}
}
FUNCTION "index" with a writer and response
ExecuteTemplate takes a response "and" an error type
so use it! to check for an error.
